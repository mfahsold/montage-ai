STABILIZE_AI=true
├─ Enables KI stabilization (required for any mode)

STABILIZE_MODE=extreme
├─ PROFILE_EXTREME: Aggressive (4-pass deshake, 0.85 smooth)
├─ PROFILE_SUPER_EXTREME: Ultra (4-pass deshake, 0.95 smooth) ⭐
├─ PROFILE_DOCUMENTARY: Moderate (2-pass deshake, 0.5 smooth)
└─ PROFILE_BROADCAST: Balanced (2-pass deshake, 0.4 smooth)

AGGRESSIVE_SMOOTHING=true
├─ Enables minterpolate: 1.5x FPS → 3.5x FPS (210fps base)
└─ +233% more frame interpolation for cinema-smooth motion

FAST_STABILIZATION=true
├─ Skips minterpolate (2x faster)
└─ Use when time > quality

SKIP_COLOR_CORRECTION=true
├─ Disables color grading
└─ Preserves original colors if needed

📊 FILTER CHAIN (11 stages):
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

1. vidstabtransform  → Primary stabilization (shakiness=10, accuracy=15)
2. deshake          → Micro-stabilization (4 iterations)
3. deflicker        → Temporal coherence
4. minterpolate ⭐   → 3.5x FPS interpolation (210fps, AOBMC)
5. nlmeans          → Aggressive denoising (sigma 4.0-5.0)
6. normalize        → Adaptive contrast boost
7. curves           → Shadow lift
8. unsharp          → Sharpening (amount 1.8)
9. colorlevels      → Black crush + highlight preservation
10. saturate        → +35% saturation
11. colorbalance    → Warm cinematic grading + teal/orange
